let currentMode = 'stream'; // 'stream' or 'process'

document.addEventListener('DOMContentLoaded', () => {
renderList();
});

// 탭 전환 함수
function switchTab(mode) {
currentMode = mode;

// 버튼 스타일 변경
document.getElementById('btn-tab-stream').classList.toggle('active', mode === 'stream');
document.getElementById('btn-tab-process').classList.toggle('active', mode === 'process');

// 뷰 전환
document.getElementById('view-stream').style.display = mode === 'stream' ? 'block' : 'none';
document.getElementById('view-process').style.display = mode === 'process' ? 'block' : 'none';

renderList();
}

// 왼쪽 리스트 렌더링 함수
function renderList() {
const container = document.getElementById('list-container');
container.innerHTML = '';

if (currentMode === 'stream') {
for (let i = 1; i <= 21; i++) {
const btn = document.createElement('button');
btn.className = 'item-btn';
btn.innerText = [${i}] ${streamData[i].name};
btn.onclick = () => updateStreamView(i, btn);
container.appendChild(btn);
}
updateStreamView(13, container.children[12]); // 기본 13번 선택
} else {
for (let i = 1; i <= 12; i++) {
const btn = document.createElement('button');
btn.className = 'item-btn';
btn.innerText = Step ${i}. ${processData[i].name.split(' (')[0]};
btn.onclick = () => updateProcessView(i, btn);
container.appendChild(btn);
}
updateProcessView(4, container.children[3]); // 기본 4번(SMR) 선택
}
}

function formatNumber(num, decimals = 2) {
return num.toLocaleString('en-US', { minimumFractionDigits: decimals, maximumFractionDigits: decimals });
}

// 스트림 뷰 업데이트
function updateStreamView(id, clickedBtn) {
document.querySelectorAll('.item-btn').forEach(b => b.classList.remove('active'));
if(clickedBtn) clickedBtn.classList.add('active');

const data = streamData[id];
document.getElementById('stream-title').innerText = Stream ${id} : ${data.name};
document.getElementById('val-temp').innerHTML = ${formatNumber(data.temp, 1)} <span class="prop-unit">°C</span>; document.getElementById('val-press').innerHTML = ``${formatNumber(data.press, 2)} &lt;span class="prop-unit"&gt;MPa&lt;/span>;
document.getElementById('val-mass').innerHTML = ${formatNumber(data.mass, 0)} <span class="prop-unit">kg/hr</span>; document.getElementById('val-enth').innerHTML = ``${formatNumber(data.enth, 2)} &lt;span class="prop-unit"&gt;GJ/hr&lt;/span>;

const tbody = document.getElementById('mole-table-body');
tbody.innerHTML = '';
for (const [formula, val] of Object.entries(data.mols)) {
const tr = document.createElement('tr');
let valClass = val === 0 ? 'zero-val' : (val > 0.5 ? 'highlight-val' : '');
tr.innerHTML = &lt;td&gt;${compNames[formula]}</td>
<td style="font-family: monospace; font-weight: bold;">${formula.replace(/(\d+)/g, '&lt;sub&gt;$1</sub>')}</td>
<td class="${valClass}"&gt;${val.toFixed(4)}</td>
`;
tbody.appendChild(tr);
}
}

// 공정 뷰 업데이트
function updateProcessView(id, clickedBtn) {
document.querySelectorAll('.item-btn').forEach(b => b.classList.remove('active'));
if(clickedBtn) clickedBtn.classList.add('active');

const data = processData[id];
document.getElementById('process-title').innerText = Step${id} : ${data.name}`;
document.getElementById('proc-desc').innerHTML = data.desc;
document.getElementById('proc-chem').innerHTML = data.chem;
document.getElementById('proc-energy').innerHTML = data.energy;
}
